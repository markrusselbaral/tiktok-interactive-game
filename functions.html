🌹 Rose → pick a fighter
handleViewerGift(user, count, giftName, diamondCount) ROSE ENTRY POINT
Handles a viewer's gift (rose). First rose → picks a fighter for that viewer (auto-balanced to the underdog side). Subsequent roses → fuel the ultimate, rate-limited per viewer. 
handleViewerGift(user, count, giftName, diamondCount) {
  if (!user) return
  this._pushRecent({
    kind: 'gift',
    user,
    text: `${giftName || 'gift'} ×${count}${diamondCount ? ` (${diamondCount}💎)` : ''}`,
  })

  let side = this.viewerSides[user]
  if (!side) {
    // first rose → pick a fighter (auto-balance to the side with fewer fans)
    side = this._underdogSide()
    this._assignSide(user, side)
    return
  }
  // already a fan → roses fuel the ultimate (rate-limited)
  if (this._cooldownOk(this._ultimateCooldown, user, ULTIMATE_COOLDOWN_MS)) {
    this.triggerUltimate(side)
  }
}

_underdogSide() SUPPORT
Picks the side with fewer fans, used by the first-rose auto-balance.
_underdogSide() {
  return this.live.fanCount.a <= this.live.fanCount.b ? 'a' : 'b'
}
_assignSide(user, side) SUPPORT
Assigns a viewer to a fighter side, updates fan counts, and emits a "joined team" feed entry. 
_assignSide(user, side) {
  if (!user || !side) return
  const previous = this.viewerSides[user]
  if (previous === side) return
  if (previous)
    this.live.fanCount[previous] = Math.max(0, this.live.fanCount[previous] - 1)
  this.viewerSides[user] = side
  this.live.fanCount[side] = (this.live.fanCount[side] || 0) + 1
  const name = side === 'a' ? this.a.cat.name : this.b.cat.name
  this._pushRecent({ kind: 'pick', user, text: `joined team ${name}` })
}

triggerUltimate(side) FIRED BY ROSES
Plays the meme overlay ultimate for the side. Triggered by repeat roses from a fan. 
triggerUltimate(side) {
  const f = side === 'a' ? this.a : this.b
  if (f.status === 'eliminated') return
  const name = f.evolved ? f.cat.evolvedName : f.cat.name
  this.ultimate = {
    side,
    key: Date.now(),
    name,
    ultName: `${name.toUpperCase()} ULTIMATE!`,
    meme: f.cat.meme,
    color: f.evolved ? f.cat.evolvedColor : f.cat.color,
  }
}
Viewer comments & targeting
handleViewerComment(user, text)
Parses chat. "a kick" → targeted skill. Bare side mention → join team. Bare skill → use viewer's team (auto-joins underdog if none). 
handleViewerComment(user, text) {
  if (!user) return
  const trimmed = String(text || '').trim()
  this._pushRecent({ kind: 'comment', user, text: trimmed })

  const hinted = detectSide(trimmed)
  const skillAmount = detectSkill(trimmed)

  if (hinted && skillAmount) {
    if (this._cooldownOk(this._skillCooldown, user, SKILL_COOLDOWN_MS)) {
      this.addScore(hinted, skillAmount)
    }
    return
  }
  if (hinted) { this._assignSide(user, hinted); return }
  if (skillAmount) {
    let side = this.viewerSides[user]
    if (!side) {
      this._assignSide(user, this._underdogSide())
      side = this.viewerSides[user]
    }
    if (this._cooldownOk(this._skillCooldown, user, SKILL_COOLDOWN_MS)) {
      this.addScore(side, skillAmount)
    }
  }
}
Combat actions
addScore(side, amount)
Applies damage, picks attack animation (punch / kick / combo), and triggers a delayed hurt reaction on the opponent.
addScore(side, amount) {
  const f = side === 'a' ? this.a : this.b
  if (f.status === 'eliminated') return
  f.score += amount
  this.flash = { side, amount, key: Date.now() }
  const type = amount >= 10 ? 'combo' : amount >= 5 ? 'kick' : 'punch'
  this.attack[side] = { key: Date.now(), type }
  const other = side === 'a' ? 'b' : 'a'
  setTimeout(() => { this.hurt[other] = Date.now() }, 180)
}
executeUltimateFollowup(side)
Dramatic on-stage combo after the meme overlay exits: punch → kick → punch → kick → finishing combo.
executeUltimateFollowup(side) {
  const f = side === 'a' ? this.a : this.b
  if (f.status === 'eliminated') return
  const sequence = [
    { delay: 50,   amount: 3 },   // punch
    { delay: 380,  amount: 5 },   // kick
    { delay: 760,  amount: 3 },   // punch
    { delay: 1140, amount: 5 },   // kick
    { delay: 1600, amount: 12 },  // combo finisher
  ]
  sequence.forEach(({ delay, amount }) => {
    setTimeout(() => {
      if (f.status === 'eliminated') return
      this.addScore(side, amount)
    }, delay)
  })
}
clearUltimate()
Clears the ultimate overlay and chains into the followup combo.
clearUltimate() {
  const side = this.ultimate?.side
  this.ultimate = null
  if (side) this.executeUltimateFollowup(side)
}
revive(side)
Restores an eliminated fighter to fighting status.
revive(side) {
  const f = side === 'a' ? this.a : this.b
  f.status = 'fighting'
  f.score = Math.max(f.score, 1)
  this.lastEvent = { type: 'revive', side, key: Date.now() }
}
evolve(side)
Toggles a fighter into their evolved form (e.g., Tangerine → Inferno Tangerine).
evolve(side) {
  const f = side === 'a' ? this.a : this.b
  f.evolved = !f.evolved
  this.lastEvent = { type: 'evolve', side, key: Date.now() }
}
Round flow
startTimer() / stopTimer()
Per-second countdown that holds until the first point is scored.
startTimer() {
  this.stopTimer()
  this._timer = setInterval(() => {
    if (!this.roundActive) return
    if (this.a.score === 0 && this.b.score === 0) return
    if (this.timeLeft > 0) this.timeLeft -= 1
    else this.endRound()
  }, 1000)
}

stopTimer() {
  if (this._timer) { clearInterval(this._timer); this._timer = null }
}
endRound()
Scores the round, eliminates the loser, queues the winner-highlight video.
endRound() {
  if (!this.roundActive) return
  this.roundActive = false
  const aWins = this.a.score >= this.b.score
  const winner = aWins ? this.a : this.b
  const loser  = aWins ? this.b : this.a
  const winnerSide = aWins ? 'a' : 'b'
  const loserSide  = aWins ? 'b' : 'a'
  winner.status = 'winner'
  loser.status  = 'eliminated'
  this.eliminated.push(loser.cat.id)
  this.lastEvent = { type: 'win', side: winnerSide, key: Date.now() }
  setTimeout(() => {
    this.lastEvent = { type: 'eliminate', side: loserSide, key: Date.now() }
  }, 250)
  if (winner.cat.video) {
    this.winnerHighlight = {
      side: winnerSide, key: Date.now(),
      name: winner.evolved ? winner.cat.evolvedName : winner.cat.name,
      video: winner.cat.video,
      color: winner.evolved ? winner.cat.evolvedColor : winner.cat.color,
    }
  }
}

nextRound()
Brings in the next challenger from the queue (recycling eliminated cats if empty), replaces the loser slot, resets scores and fan counts. 
nextRound() {
  this.winnerHighlight = null
  let nextId = null
  while (this.queue.length) {
    const candidate = this.queue.shift()
    if (!this.eliminated.includes(candidate)) { nextId = candidate; break }
  }
  if (!nextId) {
    const pool = this.roster.map((c) => c.id)
      .filter((id) => id !== this.a.cat.id && id !== this.b.cat.id)
    if (pool.length) {
      this.eliminated = []
      this.queue = pool.slice(1)
      nextId = pool[0]
    }
  }
  // …replace loser slot with challenger, reset round state, _resetFanship()
}

Sets the round timer length; updates the active timer if a round is running.
setRoundDuration(seconds) {
  this.roundDuration = seconds
  if (this.roundActive) this.timeLeft = seconds
}
Live (TikTok) integration
connectLive(username)
Opens a WebSocket to the bridge server with the supplied TikTok username.
connectLive(username) {
  const trimmed = String(username || '').replace(/^@/, '').trim()
  if (!trimmed) {
    this.live.status = 'error'
    this.live.error  = 'Enter a TikTok username'
    return
  }
  this.disconnectLive()
  this.live.username = trimmed
  this.live.status = 'connecting'
  const url = `${this.live.bridgeUrl}/?username=${encodeURIComponent(trimmed)}`
  const ws = new WebSocket(url)
  this._ws = ws
  ws.addEventListener('message', (m) => {
    try { this._handleLiveEvent(JSON.parse(m.data)) } catch {}
  })
  ws.addEventListener('close', () => { if (this.live.status !== 'error') this.live.status = 'closed' })
  ws.addEventListener('error', () => { this.live.status = 'error' })
}
disconnectLive()
Closes the bridge WebSocket and resets live status to idle.
disconnectLive() {
  if (this._ws) { try { this._ws.close() } catch {}; this._ws = null }
  this.live.status = 'idle'
}
_handleLiveEvent(evt)
Routes inbound bridge events: connected / disconnected / error / comment / gift.
_handleLiveEvent(evt) {
  switch (evt.type) {
    case 'connected':    this.live.status = 'connected'; break
    case 'disconnected':
    case 'streamEnd':    this.live.status = 'closed'; break
    case 'error':        this.live.status = 'error'; break
    case 'comment':      this.handleViewerComment(evt.user, evt.text); break
    case 'gift':         this.handleViewerGift(evt.user, evt.count || 1, evt.giftName, evt.diamondCount); break
  }
}
Internal helpers
_cooldownOk(map, user, ms)
Per-user rate limiter shared by skill and ultimate.
_cooldownOk(map, user, ms) {
  const now = Date.now()
  const last = map[user] || 0
  if (now - last < ms) return false
  map[user] = now
  return true
}
_resetFanship()
Wipes viewer→side map, fan counts, and cooldowns when a new round begins.
_resetFanship() {
  this.viewerSides = {}
  this.live.fanCount = { a: 0, b: 0 }
  this._skillCooldown = {}
  this._ultimateCooldown = {}
}
_pushRecent(entry)
Prepends an event to the live feed (capped at MAX_RECENT_EVENTS).
_pushRecent(entry) {
  this.live.recent.unshift({ key: Date.now() + Math.random(), ...entry })
  if (this.live.recent.length > MAX_RECENT_EVENTS) {
    this.live.recent.length = MAX_RECENT_EVENTS
  }
}
Module-level helpers
detectSide(text)
Maps free-text to side 'a' / 'b' via SIDE_HINTS (orange/tangerine/inferno → a, white/marshmallow/frostbite → b).
function detectSide(text) {
  const lower = String(text || '').toLowerCase()
  for (const side of ['a', 'b']) {
    for (const hint of SIDE_HINTS[side]) {
      const re = new RegExp(`(?:^|[^a-z])${hint}(?:[^a-z]|$)`, 'i')
      if (re.test(lower)) return side
    }
  }
  return null
}
detectSkill(text)
Maps a comment to skill damage: combo/super/special → 10, kick/sweep → 5, punch/jab/hit/smack → 3.
function detectSkill(text) {
  for (const { re, amount } of SKILL_PATTERNS) {
    if (re.test(text)) return amount
  }
  return null
}
makeFighter(cat)
Factory for a fresh fighter slot.
function makeFighter(cat) {
  return { cat, score: 0, status: 'fighting', evolved: false }
